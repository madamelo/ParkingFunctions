\begin{example}[Définition 3 : $n = 4$]
    ~
    \begin{itemize}
        \item $f_1 = (1, 2, 2, 3) \in \mathcal{PF'}_4$
        \item $f_2 = (1, 2, 3, 2) \notin \mathcal{PF'}_4
         \text{, bien que } f_2 \in \mathcal{PF}_4$
    \end{itemize}
\end{example}
