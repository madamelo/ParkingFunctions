\begin{theorem}
    Let $nc_{a,b}$ be the cardinal of $\mathcal{NC}_{a,b}$.
    We have $$nc_{a,b} = \frac{1}{a+b} \binom{a+b}{a} = 
    \frac{(a+b-1)!}{a!b!}$$, which is the rational Catalan
    number.
\end{theorem}

\begin{example}
    all abncp for some a b
\end{example}

\end{proof}
